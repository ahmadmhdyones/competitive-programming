/* Magic Numbers: https://codeforces.com/contest/320/problem/A */

#include <bits/stdc++.h>

    string s;
    cin >> s;
    cout << (regex_match(s,regex("(1|14|144)*")) ? "YES" : "NO");

}

/* Another Solution */

#include <iostream>

using namespace std;

int i;

int main ()
{
	string s;
	cin >> s;
	for ( i = 0; i < s.size(); i ++) {
		if ( s[i] == '1' && s[i+1] == '4' && s[i+2] == '4' ) { i+=2 ;continue; }
		if ( s[i] == '1' && s[i+1] == '4' ) { i++; continue; }
		if ( s[i] == '1' ) continue;
		break;
	}
	if ( i == s.size() )
		cout << "YES" << endl;
	else
		cout << "NO" << endl;
	return 0;
}
