/* Diverse Strings: https://codeforces.com/contest/1144/problem/A */

#include <bits/stdc++.h>

using namespace std;

#define forn(i, n) for (int i = 0; i < int(n); i++)

int main() {
    int n;
    cin >> n;
    forn(i, n) {
        string s;
        cin >> s;
        if (set<char>(s.begin(), s.end()).size() == s.length()
                && *max_element(s.begin(), s.end()) == char(*min_element(s.begin(), s.end()) + (s.length() - 1)))
            cout << "Yes" << endl;
        else
            cout << "No" << endl;
    }
}

/* Another Solution */

#include <bits/stdc++.h>

using namespace std;

int main ()
{
     int n;
     string s;
     bool ok;
     cin >> n;
     while ( n -- ) {
          ok = false;
          cin >> s;
          sort(s.begin(),s.end());
          for ( int i = 0 ; i < s.size()-1; i ++)
               if ( s[i] != s[i+1]-1 ) {
                    cout << "NO" << endl;
                    ok = true;
                    break;
               }
          if ( !ok ) cout << "YES" << endl;
     }

     return 0;
}
