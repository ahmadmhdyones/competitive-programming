- We can Compare between two strings by one of the following methods:

A. Char by Char:
________________
#include <bits/stdc++.h>
using namespace std;
string f, s;
int main ()
{
	cin >> f >> s;
	for ( int i = 0; i < max(s.size(),f.size()); i ++ ) 
		if ( tolower(s[i]) < tolower(f[i]) ) return cout << 1 << endl, 0;
		else if ( tolower(s[i]) > tolower(f[i]) ) return cout << -1 << endl, 0;
	return cout << 0 << endl, 0;
}

C. Operators:
_____________
#include <iostream>
using namespace std;
int main()
{
    string a,b;
    cin>>a>>b;
    int n=a.length();
    for (int i=0;i<n;i++)
        a[i]=tolower(a[i]), b[i]=tolower(b[i]);
    if (a<b) cout<<-1<<endl;
    else if (b<a) cout<<1<<endl;
    else cout<<0<<endl;
}
